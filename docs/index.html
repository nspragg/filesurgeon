Manipulate files the easy way

Common examples

const FileSurgeon = require("filesurgeon");

FileSurgeon.asStream("myFile")
  .prepend("a new line")
  .append("a new line")
  .replace("x", "y")
  .map(makeSomeChange())
  .run();
